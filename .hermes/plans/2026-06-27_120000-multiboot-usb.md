# MultiBoot USB — Lightweight Secure Multi-ISO Bootloader

> **For Hermes:** Use subagent-driven-development skill to implement this plan task-by-task.

**Goal:** A Linux-first Python tool that partitions a USB drive, installs GRUB2 bootloader on a dedicated EFI partition, and auto-discovers ISO files on a separate storage partition — presenting a graphical boot menu at startup.

**Architecture:** GPT disk with 3 partitions: (1) EFI System Partition with GRUB EFI binary + grub.cfg, (2) boot partition with GRUB modules/themes/memdisk, (3) ISO storage partition (FAT32 default, ext4/exFAT optional). GRUB uses `loopback` + `iso9660` to chainload ISO kernels. A Python setup script handles partitioning, GRUB installation, ISO metadata management, and grub.cfg generation. At boot, GRUB displays a graphical menu auto-generated from the ISO partition contents.

**Tech Stack:** Python 3.12+, `parted`/`sgdisk` for partitioning, `grub-install` for GRUB2, `mkfs.fat`/`mkfs.ext4` for formatting, `cryptography` for ISO hash verification, `tkinter` for GUI metadata editor, `pyinstaller` for Windows port.

---

## Design Decisions

| Decision | Choice | Rationale |
|----------|--------|-----------|
| Partition scheme | GPT | Modern, required for UEFI, supports >2TB |
| Boot partition | FAT32 (ESP) | UEFI standard, universally readable |
| ISO partition | FAT32 default, ext4/exFAT optional | Most ISOs fit DVD size (~4.7GB); ext4 for >4GB; exFAT for Windows compat |
| Boot mode | UEFI primary, optional BIOS | Security-first; BIOS fallback via MBR gap embedding |
| GRUB config | Graphical menu with icons | User-friendly; per-distro icons in themes dir |
| ISO discovery | Auto-scan on setup + re-scan on demand | Drop ISO → run `multiboot-usb refresh` or auto-detect at next boot |
| Security | SHA256 sidecar files, optional Secure Boot signing | Verify ISO integrity before boot |
| Metadata | Auto-detected from ISO (distro, version) + GUI editor | Minimal user effort; override via tkinter GUI |

```python
"""Disk partitioning and formatting for multiboot-usb."""
import subprocess
import shlex
from dataclasses import dataclass
from pathlib import Path
from typing import List


@dataclass
class PartitionScheme:
    """Creates a GPT layout with 3 partitions on a USB drive."""
    device: str
    efi_size: str = "100M"       # EFI System Partition
    boot_size: str = "500M"      # GRUB modules/themes
    # ISO partition uses remaining space

    def get_commands(self) -> List[str]:
        """Return list of shell commands to create the partition scheme."""
        return [
            # Create GPT, create 3 partitions
            f"sgdisk -Z {self.device}",
            f"sgdisk -a 2048 -og {self.device}",
        ]

    def partition(self) -> dict:
        """Execute partitioning and return partition paths."""
        # Wipe and create GPT
        subprocess.run(["sgdisk", "-Z", self.device], check=True)
        subprocess.run(["sgdisk", "-a", "2048", "-og", self.device], check=True)

        # Create partitions using parted (more reliable for ESP flag)
        parted_cmds = [
            "mklabel gpt",
            f"mkpart primary fat32 1MiB {self.efi_size} set 1 esp on",
            f"mkpart primary fat32 {self.efi_size} {self._boot_end()} name 2 boot",
            f"mkpart primary fat32 {self._boot_end()} 100% name 3 isos",
        ]
        for cmd in parted_cmds:
            subprocess.run(["parted", "--script", self.device, cmd], check=True)

        # Wait for kernel to register partitions
        subprocess.run(["partprobe", self.device], check=True)

        return {
            "efi": f"{self.device}1",
            "boot": f"{self.device}2",
            "isos": f"{self.device}3",
        }

    def _boot_end(self) -> str:
        # Parse efi_size and add boot_size
        efi_mb = int(self.efi_size.rstrip("Mm"))
        boot_mb = int(self.boot_size.rstrip("Mm"))
        return f"{efi_mb + boot_mb}MiB"


def format_partition(device: str, fs_type: str, label: str) -> None:
    """Format a partition with the given filesystem type and label."""
    if fs_type == "fat32":
        cmd = ["mkfs.fat", "-F", "32", "-n", label, device]
    elif fs_type == "exfat":
        cmd = ["mkfs.exfat", "-n", label, device]
    elif fs_type == "ext4":
        cmd = ["mkfs.ext4", "-L", label, device]
    else:
        raise ValueError(f"Unsupported filesystem: {fs_type}")

    subprocess.run(cmd, check=True)


def detect_usb_device(name_fragment: str) -> str:
    """Find USB device by name/size hint. Returns /dev/sdX path."""
    result = subprocess.run(
        ["lsblk", "-d", "-o", "NAME,TRAN,SIZE", "-n", "-p"],
        capture_output=True, text=True, check=True
    )
    for line in result.stdout.strip().split("\n"):
        parts = line.split()
        if len(parts) >= 3 and parts[1] == "usb" and name_fragment.lower() in parts[2].lower():
            return parts[0]
    raise RuntimeError(f"No USB device matching '{name_fragment}' found")
```

```python
"""GRUB2 installation for multiboot-usb."""
import subprocess
import shutil
from pathlib import Path
from typing import Optional


class GrubInstaller:
    """Installs GRUB2 to EFI System Partition and optionally MBR gap."""

    GRUB_MODULES = [
        "normal", "linux", "loopback", "iso9660", "gzio", "fat",
        "part_gpt", "part_msdos", "efi_gop", "gfxterm", "gfxmenu",
        "png", "boot", "configfile", "search", "search_fs_uuid",
    ]

    def __init__(self, device: str, efi_partition: str, boot_partition: str,
                 mount_point: str, enable_bios: bool = False):
        self.device = device
        self.efi_partition = efi_partition
        self.boot_partition = boot_partition
        self.mount_point = Path(mount_point)
        self.enable_bios = enable_bios
        self.boot_dir = self.mount_point / "boot"
        self.grub_dir = self.boot_dir / "grub"

    def install(self) -> None:
        """Install GRUB2 for UEFI and optionally BIOS."""
        self._prepare_boot_dir()
        self._install_efi()
        if self.enable_bios:
            self._install_bios()
        self._copy_modules()

    def _prepare_boot_dir(self) -> None:
        """Create boot directory structure."""
        self.grub_dir.mkdir(parents=True, exist_ok=True)
        (self.boot_dir / "iso").mkdir(exist_ok=True)
        (self.boot_dir / "metadata").mkdir(exist_ok=True)

    def _install_efi(self) -> None:
        """Install GRUB EFI binary to ESP."""
        subprocess.run([
            "grub-install",
            "--target=x86_64-efi",
            "--efi-directory", str(self.mount_point),
            "--boot-directory", str(self.boot_dir),
            "--removable",
            "--recheck",
            self.device,
        ], check=True)

    def _install_bios(self) -> None:
        """Install GRUB to MBR gap for BIOS fallback."""
        subprocess.run([
            "grub-install",
            "--target=i386-pc",
            "--boot-directory", str(self.boot_dir),
            "--recheck",
            self.device,
        ], check=True)

    def _copy_modules(self) -> None:
        """Copy required GRUB modules from system."""
        system_grub = Path("/usr/lib/grub/x86_64-efi")
        if not system_grub.exists():
            system_grub = Path("/usr/lib/grub2/x86_64-efi")
        if not system_grub.exists():
            raise FileNotFoundError("GRUB EFI modules not found on system")

        target = self.grub_dir / "x86_64-efi"
        target.mkdir(exist_ok=True)

        for module in system_grub.glob("*.mod"):
            if module.stem in self.GRUB_MODULES:
                shutil.copy2(module, target)

        # Copy required files
        for fname in ["grubenv", "fonts", "locale"]:
            src = system_grub.parent / fname
            if src.exists():
                dst = self.grub_dir / fname
                if src.is_dir():
                    if dst.exists():
                        shutil.rmtree(dst)
                    shutil.copytree(src, dst)
```

```python
"""ISO scanning and metadata auto-detection for multiboot-usb."""
import hashlib
import json
import re
import subprocess
from dataclasses import dataclass, asdict
from pathlib import Path
from typing import List, Optional


@dataclass
class IsoMetadata:
    filename: str
    distro: str
    version: str
    arch: str
    boot_type: str  # "bios", "uefi", "both"
    size: int
    sha256: str
    menu_label: str
    custom_label: Optional[str] = None

    @property
    def display_name(self) -> str:
        return self.custom_label or self.menu_label

    def to_dict(self) -> dict:
        return asdict(self)

    @classmethod
    def from_dict(cls, data: dict) -> "IsoMetadata":
        return cls(**data)


class IsoScanner:
    """Scans a directory for bootable ISO files and extracts metadata."""

    # Patterns for common distro naming
    DISTRO_PATTERNS = [
        (r"ubuntu-(\d+\.\d+)", "ubuntu"),
        (r"debian-(\d+\.\d+)", "debian"),
        (r"archlinux-(\d{4}\.\d{2}\.\d{2})", "arch"),
        (r"fedora-(\d+)", "fedora"),
        (r"linuxmint-(\d+\.\d+)", "mint"),
        (r"opensuse-(\d+\.\d+)", "opensuse"),
        (r"manjaro-(\d+\.\d+)", "manjaro"),
        (r"centos-(\d+)", "centos"),
        (r"alma(\d+)", "almalinux"),
        (r"rocky-(\d+)", "rocky"),
        (r"pop_os-(\d+\.\d+)", "pop"),
        (r"elementary-(\d+\.\d+)", "elementary"),
        (r"zorin-(\d+)", "zorin"),
        (r"mx-linux-(\d+)", "mx"),
        (r"kali-(\d{4}\.\d+)", "kali"),
    ]

    def __init__(self, iso_directory: str):
        self.iso_dir = Path(iso_directory)

    def scan(self) -> List[IsoMetadata]:
        """Scan directory for ISO files and return metadata list."""
        results = []
        for iso_path in sorted(self.iso_dir.glob("*.iso")):
            meta = self._build_metadata(iso_path)
            results.append(meta)
        return results

    def _build_metadata(self, iso_path: Path) -> IsoMetadata:
        """Build metadata for a single ISO."""
        filename = iso_path.name
        distro, version = self._detect_metadata(filename)
        arch = self._detect_arch(filename)
        sha256 = self._compute_hash(iso_path)
        boot_type = self._detect_boot_type(iso_path)

        label = f"{distro.title()} {version}" if distro != "unknown" else filename

        return IsoMetadata(
            filename=filename,
            distro=distro,
            version=version,
            arch=arch,
            boot_type=boot_type,
            size=iso_path.stat().st_size,
            sha256=sha256,
            menu_label=label,
        )

    def _detect_metadata(self, filename: str) -> tuple[str, str]:
        """Auto-detect distro and version from filename."""
        lower = filename.lower()
        for pattern, distro_name in self.DISTRO_PATTERNS:
            match = re.search(pattern, lower)
            if match:
                return distro_name, match.group(1)
        return "unknown", ""

    def _detect_arch(self, filename: str) -> str:
        """Detect architecture from filename."""
        lower = filename.lower()
        if "amd64" in lower or "x86_64" in lower or "x64" in lower:
            return "amd64"
        elif "i386" in lower or "i686" in lower:
            return "i386"
        elif "arm64" in lower or "aarch64" in lower:
            return "arm64"
        return "unknown"

    def _detect_boot_type(self, iso_path: Path) -> str:
        """Check if ISO supports UEFI, BIOS, or both via eltorito inspection."""
        try:
            result = subprocess.run(
                ["file", str(iso_path)],
                capture_output=True, text=True, timeout=10
            )
            output = result.stdout.lower()
            if "uefi" in output or "efi" in output:
                return "both" if "bootable" in output else "uefa"
            elif "bootable" in output:
                return "bios"
        except (subprocess.TimeoutExpired, FileNotFoundError):
            pass
        return "both"  # Default to both for safety

    def _compute_hash(self, iso_path: Path) -> str:
        """Compute SHA256 hash of ISO file."""
        sha256 = hashlib.sha256()
        with open(iso_path, "rb") as f:
            for chunk in iter(lambda: f.read(8192), b""):
                sha256.update(chunk)
        return sha256.hexdigest()

    def save_metadata(self, metadata: IsoMetadata, directory: str) -> None:
        """Save metadata as JSON sidecar file."""
        meta_dir = Path(directory) / "metadata"
        meta_dir.mkdir(parents=True, exist_ok=True)
        meta_file = meta_dir / f"{metadata.filename}.json"
        with open(meta_file, "w") as f:
            json.dump(metadata.to_dict(), f, indent=2)

    def load_metadata(self, iso_filename: str, directory: str) -> IsoMetadata:
        """Load metadata from JSON sidecar file."""
        meta_file = Path(directory) / "metadata" / f"{iso_filename}.json"
        with open(meta_file) as f:
            data = json.load(f)
        return IsoMetadata.from_dict(data)
```

"""GRUB configuration file generator for multiboot-usb."""
import subprocess
from pathlib import Path
from typing import List

from multiboot_usb.iso_scanner import IsoMetadata


class GrubConfigGenerator:
    """Generates grub.cfg with graphical menu for ISO boot entries."""

    TEMPLATE_HEADER = """# multiboot-usb GRUB configuration
# Auto-generated — do not edit manually

set timeout=10
set default=0
set gfxmode=auto
set gfxpayload=keep

insmod gfxterm
insmod png
insmod part_gpt
insmod iso9660
insmod loopback

# Load distro icons if available
if [ -f ${prefix}/themes/multiboot/icons.png ]; then
    background_image ${prefix}/themes/multiboot/icons.png
fi

menuentry "Reboot" --class reboot {
    exit
}

menuentry "Shutdown" --class halt {
    halt
}

"""

    TEMPLATE_ENTRY = """menuentry "{label}" --class {distro} {{
    set isofile=/boot/iso/{filename}
    loopback loop $isofile
    linux (loop)/{kernel_path} {kernel_args} iso-scan/filename=$isofile
    initrd (loop)/{initrd_path}
    boot
}}

"""

    # Distro-specific kernel path detection
    DISTRO_KERNEL_MAP = {
        "ubuntu": {
            "kernel": "casper/vmlinuz",
            "initrd": "casper/initrd",
            "args": "boot=casper quiet splash --",
        },
        "debian": {
            "kernel": "install.amd/vmlinuz",
            "initrd": "install.amd/initrd.gz",
            "args": "quiet --",
        },
        "arch": {
            "kernel": "arch/boot/x86_64/vmlinuz-linux",
            "initrd": "arch/boot/x86_64/archiso.img",
            "args": "img_loop=/arch/x86_64.img earlymodules=loop",
        },
        "fedora": {
            "kernel": "isolinux/vmlinuz",
            "initrd": "isolinux/initrd.img",
            "args": "root=live:CDLABEL=LABEL quiet rd.live.image iso-scan/filename=${isofile}",
        },
        "mint": {
            "kernel": "casper/vmlinuz",
            "initrd": "casper/initrd",
            "args": "boot=casper quiet splash --",
        },
        "opensuse": {
            "kernel": "boot/x86_64/loader/linux",
            "initrd": "boot/x86_64/loader/initrd",
            "args": "isofrom_device=/dev/disk/by-uuid/{uuid} isofrom_system=/boot/iso/{filename} quiet",
        },
    }

    def __init__(self, iso_partition: str, uuid: str = ""):
        self.iso_partition = iso_partition
        self.uuid = uuid or self._get_partition_uuid(iso_partition)

    def _get_partition_uuid(self, partition: str) -> str:
        """Get UUID of partition via blkid."""
        try:
            import subprocess
            result = subprocess.run(
                ["blkid", "-s", "UUID", "-o", "value", partition],
                capture_output=True, text=True, check=True
            )
            return result.stdout.strip()
        except (subprocess.CalledProcessError, FileNotFoundError):
            return "UNKNOWN"

    def generate(self, isos: List[IsoMetadata]) -> str:
        """Generate complete grub.cfg content."""
        cfg = self.TEMPLATE_HEADER

        for iso in isos:
            entry = self._generate_entry(iso)
            cfg += entry

        return cfg

    def _generate_entry(self, meta: IsoMetadata) -> str:
        """Generate a single menu entry for an ISO."""
        distro_key = meta.distro.lower()
        kernel_info = self.DISTRO_KERNEL_MAP.get(distro_key, {
            "kernel": "isolinux/vmlinuz",
            "initrd": "isolinux/initrd.img",
            "args": "quiet",
        })

        return self.TEMPLATE_ENTRY.format(
            label=meta.display_name,
            distro=meta.distro,
            filename=meta.filename,
            kernel_path=kernel_info["kernel"],
            initrd_path=kernel_info["initrd"],
            kernel_args=kernel_info["args"].format(
                uuid=self.uuid, filename=meta.filename
            ),
        )

    def write(self, isos: List[IsoMetadata], output_path: str) -> None:
        """Generate and write grub.cfg to file."""
        cfg = self.generate(isos)
        Path(output_path).parent.mkdir(parents=True, exist_ok=True)
        with open(output_path, "w") as f:
            f.write(cfg)

def cmd_setup(args):
    """Initialize USB drive with multiboot layout."""
    print(f"[*] Setting up {args.device}...")

    # 1. Partition
    scheme = PartitionScheme(args.device)
    partitions = scheme.partition()
    print(f"    EFI:  {partitions['efi']}")
    print(f"    Boot: {partitions['boot']}")
    print(f"    ISOs: {partitions['isos']}")

    # 2. Format
    mount = Path("/tmp/multiboot-usb-setup")
    mount.mkdir(exist_ok=True)

    # Format EFI as FAT32 (already done by parted, but ensure label)
    format_partition(partitions["efi"], "fat32", "EFI")
    format_partition(partitions["boot"], "fat32", "BOOT")
    format_partition(partitions["isos"], args.iso_partition_fs, "ISOS")

    # 3. Mount boot partition and install GRUB
    import subprocess
    subprocess.run(["mount", partitions["boot"], str(mount)], check=True)

    installer = GrubInstaller(
        device=args.device,
        efi_partition=partitions["efi"],
        boot_partition=partitions["boot"],
        mount_point=str(mount),
        enable_bios=args.bios,
    )
    installer.install()
    print("[+] GRUB installed.")

    # 4. Mount ISO partition and scan for existing ISOs
    iso_mount = mount / "iso"
    iso_mount.mkdir(exist_ok=True)
    subprocess.run(["mount", partitions["isos"], str(iso_mount)], check=True)

    scanner = IsoScanner(str(iso_mount))
    isos = scanner.scan()
    for iso in isos:
        scanner.save_metadata(iso, str(mount))
        print(f"[+] Found ISO: {iso.display_name}")

    # 5. Generate grub.cfg
    gen = GrubConfigGenerator(partitions["isos"])
    gen.write(isos, str(mount / "boot" / "grub" / "grub.cfg"))
    print("[+] grub.cfg generated.")

    # 6. Unmount
    subprocess.run(["umount", str(iso_mount)], check=True)
    subprocess.run(["umount", str(mount)], check=True)

    print(f"[+] Setup complete! Drop ISOs into the ISOS partition and run:")
    print(f"    multiboot-usb refresh <boot_partition_mount>")

import hashlib
import json
import subprocess
from pathlib import Path
from typing import Dict, Optional


def compute_sha256(filepath: str) -> str:
    """Compute SHA256 hash of a file."""
    sha256 = hashlib.sha256()
    with open(filepath, "rb") as f:
        for chunk in iter(lambda: f.read(8192), b""):
            sha256.update(chunk)
    return sha256.hexdigest()


def verify_iso(iso_path: str, expected_hash: str) -> bool:
    """Verify ISO against expected SHA256 hash."""
    actual = compute_sha256(iso_path)
    return actual == expected_hash

    """Generate a verification manifest for an ISO."""
    return {

        "size": Path(iso_path).stat().st_size,
        "algorithm": "sha256",

- **UEFI-first** with optional BIOS fallback
- **Auto-discovery**: Drop an ISO on the storage partition, it appears in the boot menu
- **Graphical GRUB menu** with distro icons
- **Metadata editor** (GUI) for customizing boot entries
- **ISO verification** via SHA256 sidecar files
- **Secure Boot** support (MOK enrollment)

## Quick Start

```bash
# Install
pip install -e .

# Setup USB drive (WARNING: destroys all data on /dev/sdX)
sudo multiboot-usb setup /dev/sdX

# Add an ISO
sudo multiboot-usb add-iso /mnt/usb/boot ~/Downloads/ubuntu-24.04.iso

# Edit metadata (opens GUI)
multiboot-usb edit-metadata /mnt/usb/boot

# Re-scan for new ISOs
sudo multiboot-usb refresh /mnt/usb/boot
```

## Architecture

┌──────────────────┬──────────────────┬──────────────────────────┐
│  P1: EFI (FAT32) │ P2: Boot (FAT32) │  P3: ISOs (FAT32/ext4)  │

## Security Model

1. **Partition isolation**: Bootloader and ISOs on separate partitions
2. **Read-only boot**: P1+P2 can be remounted read-only after setup
3. **ISO verification**: SHA256 hashes stored as `.json` sidecars
4. **Secure Boot**: Optional MOK-based signing chain
5. **No network**: GRUB runs fully offline — no attack surface at boot

## Requirements

- Python 3.12+
- `grub-install` (GRUB2)
- `parted`, `sgdisk`, `mkfs.fat`
- Linux host (for setup)

"""Platform abstraction layer for cross-OS compatibility."""
import sys
import shutil
from pathlib import Path


def is_windows() -> bool:
    return sys.platform == "win32"


def is_linux() -> bool:
    return sys.platform == "linux"


def get_grub_install() -> str:
    """Get platform-appropriate GRUB installer path."""
    if is_windows():
        # Look for grub-install.exe in bundled tools
        base = Path(__file__).parent.parent / "tools" / "windows"
        return str(base / "grub-install.exe")
    return "grub-install"


def get_partition_tool() -> str:
    """Get platform-appropriate partitioning tool."""
    if is_windows():
        return "diskpart"  # Or bundled linux tools via WSL
    return "parted"


def get_sgdisk_tool() -> str:
    if is_windows():
        return "sgdisk.exe"
    return "sgdisk"


def ensure_loopback_available() -> bool:
    """Check if loopback device is available (Linux only)."""
    if is_windows():
        return False  # Windows uses different mechanism
    return Path("/dev/loop0").exists()
```

## Risks & Tradeoffs

| Risk | Mitigation |
|------|-----------|
| GRUB `loopback` doesn't boot all ISOs (esp. Windows) | Document Windows exceptions; use memdisk for some |
| UEFI firmware quirks (some boards ignore ESP on removable) | Test on multiple targets; provide BIOS fallback option |
| ISO >4GB on FAT32 | Offer ext4/exFAT option; warn during setup |
| Partitioning destructive | Require explicit device path; add `--dry-run` flag |
| Secure Boot key management | Document MOK enrollment flow; make optional |

## Decisions (resolved)

1. **Windows setup**: Support both native Windows (via `diskpart` + bundled GRUB) AND WSL2 path. User chooses.
2. **ISO boot compatibility**: Build a compatibility database — map distro → kernel paths, special boot args. Extensible via JSON.
3. **Persistence**: K.I.S.S. for v1 — no persistence partition. Add later if demand exists.
4. **Network boot**: Out of scope v1.

```python
#!/usr/bin/env python3
"""Lightweight web UI for multiboot-usb configuration."""
from fastapi import FastAPI, File, UploadFile
from fastapi.staticfiles import StaticFiles
from fastapi.responses import HTMLResponse
import subprocess
import shutil
from pathlib import Path

app = FastAPI(title="multiboot-usb Config")
DATA_DIR = Path("/tmp/multiboot-usb-data")
DATA_DIR.mkdir(exist_ok=True)

@app.get("/", response_class=HTMLResponse)
async def index():
    return (Path(__file__).parent / "templates" / "index.html").read_text()

@app.get("/api/devices")
async def list_devices():
    result = subprocess.run(["lsblk", "-d", "-o", "NAME,SIZE,TRAN", "-n", "-p"],
                          capture_output=True, text=True)
    devices = []
    for line in result.stdout.strip().split("\n"):
        parts = line.split()
        if len(parts) >= 3 and parts[2] == "usb":
            devices.append({"name": parts[0], "size": parts[1]})
    return devices

@app.post("/api/setup")
async def setup(device: str):
    subprocess.run(["multiboot-usb", "setup", device], check=True)
    return {"status": "ok"}

@app.post("/api/add-iso")
async def add_iso(file: UploadFile = File(...)):
    dest = DATA_DIR / file.filename
    with open(dest, "wb") as f:
        shutil.copyfileobj(file.file, f)
    subprocess.run(["multiboot-usb", "add-iso", str(DATA_DIR), str(dest)], check=True)
    return {"status": "ok", "filename": file.filename}

if __name__ == "__main__":
    import uvicorn
    uvicorn.run(app, host="0.0.0.0", port=8080)
```
